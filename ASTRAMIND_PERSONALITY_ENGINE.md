### Automatic Tone Detection

```python
# backend/core/ASTRAMIND_ai_personality.py

def detect_required_tone(context: ConversationContext) -> ToneLevel:
    """
    Analyze conversation to determine appropriate tone
    """
    # Check if professional context
    if context.is_professional_query():
        return ToneLevel.PROFESSIONAL_CHILL
    
    # Check if celebratory moment
    if context.is_celebration():
        return ToneLevel.HYPED_ENERGETIC
    
    # Check if serious/mental health
    if context.is_serious():
        return ToneLevel.CHILL_ZEN
    
    # Check if regular conversation
    if context.conversation_history_length > 0:
        # Adapt to conversation momentum
        return detect_tone_from_history(context.messages)
    
    # Default to friendly
    return ToneLevel.CASUAL_FRIENDLY
```
